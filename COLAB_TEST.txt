# HÜCRE 6 — INFERENCE LOOP + GROUND TRUTH KARŞILAŞTIRMA
# ============================================================================
# Tüm test seti üzerinde model.forward → DB lookup → metric collection.

import time
all_results = []  # her sample için dict

print(f'Inference başlıyor ({len(test_ds)} sample, batch 64, {device})...')
t0 = time.perf_counter()

with torch.no_grad():
    for batch_idx, batch in enumerate(test_loader):
        csi = batch['csi'].to(device)
        uwb = batch['uwb'].to(device)
        link_geo = batch['link_geo'].to(device)
        slot_gt = batch['labels']['slot'].cpu().numpy()           # (B, 6)
        mat_gt = batch['labels']['material'].cpu().numpy()         # (B, 6)
        cw_gt = batch['labels']['codeword'].cpu().numpy()          # (B, 6, 7)

        out = model(csi, uwb, link_geo, return_aux=False)
        det_logits = out['detection_logits'].cpu().numpy()        # (B, 6)
        det_pred = (1 / (1 + np.exp(-det_logits)) > 0.5).astype(int)
        cw_logits = out['barcode_codeword'].cpu().numpy()         # (B, 6, 7)

        # Her sample için DB lookup
        B = csi.shape[0]
        for b in range(B):
            for s in range(6):
                if det_pred[b, s] == 0:
                    continue  # boş slot
                # Codeword decode → ID
                cw_bits = (cw_logits[b, s] > 0).astype(np.uint8)
                data_4bit, corrected = hamming_decode_7_4(cw_bits)
                pred_id = int(data_bits_to_int(data_4bit))

                # Ground truth ID
                gt_data, _ = hamming_decode_7_4(cw_gt[b, s].astype(np.uint8))
                gt_id = int(data_bits_to_int(gt_data))

                product = db.lookup(pred_id)
                all_results.append({
                    'sample_idx': batch_idx * 64 + b,
                    'slot': s,
                    'gt_slot_filled': int(slot_gt[b, s]),
                    'gt_material_id': int(mat_gt[b, s]),
                    'gt_product_id': gt_id,
                    'pred_slot_filled': int(det_pred[b, s]),
                    'pred_product_id': pred_id,
                    'hamming_corrected': bool(corrected),
                    'pred_material': product['material'],
                    'pred_name': product['name'],
                    'pred_color': product['color_hex'],
                    'gt_pred_match': pred_id == gt_id,
                })

elapsed = time.perf_counter() - t0
print(f'Inference tamamlandı: {elapsed:.1f}s ({elapsed*1000/len(test_ds):.1f} ms/sample)')
print(f'Toplam dolu slot tahmini: {len(all_results)}')
print(f'ID tahmin doğruluğu: {sum(r["gt_pred_match"] for r in all_results)}/{len(all_results)} '
      f'({100*sum(r["gt_pred_match"] for r in all_results)/max(len(all_results),1):.1f}%)')

    # r dolu slot'lar için var
    det_gt_all.append(r['gt_slot_filled'])
    det_pred_all.append(r['pred_slot_filled'])
    if r['gt_slot_filled'] == 1:
        mat_gt_dolu.append(r['gt_material_id'])
        mat_pred_dolu.append(r['pred_material'])

            hamming_corrected_count += 1

# Detection precision + recall (sadece tahmin edilen dolu slot'lar all_results'ta)
n_total_predicted_full = len(all_results)
n_true_positive = sum(1 for r in all_results if r['gt_slot_filled'] == 1)
det_precision = n_true_positive / max(n_total_predicted_full, 1)

# ID accuracy (Hamming sonrası)
n_id_correct = sum(1 for r in all_results
                    if r['gt_slot_filled'] == 1 and r['gt_pred_match'])
n_id_check_pool = sum(1 for r in all_results if r['gt_slot_filled'] == 1)
id_accuracy = n_id_correct / max(n_id_check_pool, 1)

# Material accuracy (DB lookup)
MATERIAL_NAMES = ['boş', 'metal', 'plastik', 'ahşap', 'karton']
n_mat_correct = sum(1 for r in all_results
                     if r['gt_slot_filled'] == 1 and
                     r['pred_material'] == (MATERIAL_NAMES[r['gt_material_id']]
                                             if r['gt_material_id'] < len(MATERIAL_NAMES)
                                             else 'unknown'))
mat_accuracy = n_mat_correct / max(n_id_check_pool, 1)

# Detection recall (gerçek dolu slot'ların kaçı tahmin edildi)
# Ground truth dolu slot sayısı — dataset'ten al
from data_synthesis.multi_domain_dataset import MultiDomainDataset
total_gt_filled = 0
total_gt_slots = 0
for i in range(len(test_ds)):
    s = test_ds[i]
    total_gt_filled += int(s['labels']['slot'].sum())
    total_gt_slots += 6
det_recall = n_true_positive / max(total_gt_filled, 1)

print(f'\n--- DETECTION ---')
