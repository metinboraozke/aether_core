# HÜCRE 6 — INFERENCE LOOP + GROUND TRUTH KARŞILAŞTIRMA
# ============================================================================
# Tüm test seti üzerinde model.forward → DB lookup → metric collection.

import time
all_results = []  # her sample için dict

print(f'Inference başlıyor ({len(test_ds)} sample, batch 64, {device})...')
t0 = time.perf_counter()

with torch.no_grad():
    for batch_idx, batch in enumerate(test_loader):
        csi = batch['csi'].to(device)
        uwb = batch['uwb'].to(device)
        link_geo = batch['link_geo'].to(device)
        slot_gt = batch['labels']['slot'].cpu().numpy()           # (B, 6)
        mat_gt = batch['labels']['material'].cpu().numpy()         # (B, 6)
        cw_gt = batch['labels']['codeword'].cpu().numpy()          # (B, 6, 7)

        out = model(csi, uwb, link_geo, return_aux=False)
        det_logits = out['detection_logits'].cpu().numpy()        # (B, 6)
        det_pred = (1 / (1 + np.exp(-det_logits)) > 0.5).astype(int)
        cw_logits = out['barcode_codeword'].cpu().numpy()         # (B, 6, 7)

        # Her sample için DB lookup
        B = csi.shape[0]
        for b in range(B):
            for s in range(6):
                if det_pred[b, s] == 0:
                    continue  # boş slot
                # Codeword decode → ID
                cw_bits = (cw_logits[b, s] > 0).astype(np.uint8)
                data_4bit, corrected = hamming_decode_7_4(cw_bits)
                pred_id = int(data_bits_to_int(data_4bit))

                # Ground truth ID
                gt_data, _ = hamming_decode_7_4(cw_gt[b, s].astype(np.uint8))
                gt_id = int(data_bits_to_int(gt_data))

                product = db.lookup(pred_id)
                all_results.append({
                    'sample_idx': batch_idx * 64 + b,
                    'slot': s,
                    'gt_slot_filled': int(slot_gt[b, s]),
                    'gt_material_id': int(mat_gt[b, s]),
                    'gt_product_id': gt_id,
                    'pred_slot_filled': int(det_pred[b, s]),
                    'pred_product_id': pred_id,
                    'hamming_corrected': bool(corrected),
                    'pred_material': product['material'],
                    'pred_name': product['name'],
                    'pred_color': product['color_hex'],
                    'gt_pred_match': pred_id == gt_id,
                })

elapsed = time.perf_counter() - t0
print(f'Inference tamamlandı: {elapsed:.1f}s ({elapsed*1000/len(test_ds):.1f} ms/sample)')
print(f'Toplam dolu slot tahmini: {len(all_results)}')
print(f'ID tahmin doğruluğu: {sum(r["gt_pred_match"] for r in all_results)}/{len(all_results)} '
      f'({100*sum(r["gt_pred_match"] for r in all_results)/max(len(all_results),1):.1f}%)')

    # r dolu slot'lar için var
    det_gt_all.append(r['gt_slot_filled'])
    det_pred_all.append(r['pred_slot_filled'])
    if r['gt_slot_filled'] == 1:
        mat_gt_dolu.append(r['gt_material_id'])
        mat_pred_dolu.append(r['pred_material'])

            hamming_corrected_count += 1

# Detection precision + recall (sadece tahmin edilen dolu slot'lar all_results'ta)
n_total_predicted_full = len(all_results)
n_true_positive = sum(1 for r in all_results if r['gt_slot_filled'] == 1)
det_precision = n_true_positive / max(n_total_predicted_full, 1)

# ID accuracy (Hamming sonrası)
n_id_correct = sum(1 for r in all_results
                    if r['gt_slot_filled'] == 1 and r['gt_pred_match'])
n_id_check_pool = sum(1 for r in all_results if r['gt_slot_filled'] == 1)
id_accuracy = n_id_correct / max(n_id_check_pool, 1)

# Material accuracy (DB lookup)
MATERIAL_NAMES = ['boş', 'metal', 'plastik', 'ahşap', 'karton']
n_mat_correct = sum(1 for r in all_results
                     if r['gt_slot_filled'] == 1 and
                     r['pred_material'] == (MATERIAL_NAMES[r['gt_material_id']]
                                             if r['gt_material_id'] < len(MATERIAL_NAMES)
                                             else 'unknown'))
mat_accuracy = n_mat_correct / max(n_id_check_pool, 1)

# Detection recall (gerçek dolu slot'ların kaçı tahmin edildi)
# Ground truth dolu slot sayısı — dataset'ten al
from data_synthesis.multi_domain_dataset import MultiDomainDataset
total_gt_filled = 0
total_gt_slots = 0
for i in range(len(test_ds)):
    s = test_ds[i]
    total_gt_filled += int(s['labels']['slot'].sum())
    total_gt_slots += 6
det_recall = n_true_positive / max(total_gt_filled, 1)

print(f'\n--- DETECTION ---')

def _draw_box(ax, center, size, color='gray', alpha=0.15, edge_color=None, lw=0.7):
    """Wireframe + opsiyonel fill box."""
    x, y, z = center
    dx, dy, dz = size
    if edge_color is None:
        edge_color = color
    # 12 kenar
    corners = [
        (x-dx/2, y-dy/2, z-dz/2), (x+dx/2, y-dy/2, z-dz/2),
        (x+dx/2, y+dy/2, z-dz/2), (x-dx/2, y+dy/2, z-dz/2),
        (x-dx/2, y-dy/2, z+dz/2), (x+dx/2, y-dy/2, z+dz/2),
        (x+dx/2, y+dy/2, z+dz/2), (x-dx/2, y+dy/2, z+dz/2),
    ]
    edges = [
        (0,1),(1,2),(2,3),(3,0),  # bottom
        (4,5),(5,6),(6,7),(7,4),  # top
        (0,4),(1,5),(2,6),(3,7),  # verticals
    ]
    for i, j in edges:
        xs = [corners[i][0], corners[j][0]]
        ys = [corners[i][1], corners[j][1]]
        zs = [corners[i][2], corners[j][2]]
        ax.plot(xs, ys, zs, color=edge_color, alpha=alpha*3, linewidth=lw)


def render_full_classroom(ax, products_by_slot, title='', view=(20, 35)):
    """Tüm 50 m² sınıf ortamını 3D render et.

    Args:
        ax: matplotlib 3D axes
        products_by_slot: dict {slot_idx: product_info} (dolu slot'lar)
        title: subplot başlık
        view: (elev, azim) kamera açısı
    """
    # 1) ODA wireframe
    _draw_box(ax, [ROOM_W/2, ROOM_D/2, ROOM_H/2], [ROOM_W, ROOM_D, ROOM_H],
              edge_color='#444444', alpha=0.4, lw=0.8)

    # Zemin (yarı şeffaf plane efekt — düşük opacity çizgi grid)
    for x_g in np.arange(0, ROOM_W+0.01, 1.0):
        ax.plot([x_g, x_g], [0, ROOM_D], [0, 0], color='#1a1a1a', alpha=0.3, lw=0.3)
    for y_g in np.arange(0, ROOM_D+0.01, 1.0):
        ax.plot([0, ROOM_W], [y_g, y_g], [0, 0], color='#1a1a1a', alpha=0.3, lw=0.3)

    # 2) RAF wireframe (dış kenar)
    _draw_box(ax, SHELF_CENTER, SHELF_SIZE,
              edge_color='#888888', alpha=0.5, lw=1.0)

    # 3) 6 SLOT wireframe
    for s_idx, center in enumerate(SLOT_CENTERS):
        _draw_box(ax, center, SLOT_SIZE,
                  edge_color='#4caf50', alpha=0.25, lw=0.6)

    # 4) DOLU SLOT'LAR — DB color_hex + ürün adı
    for s_idx, info in products_by_slot.items():
        center = SLOT_CENTERS[s_idx]
        color = info.get('color_hex', '#999999')
        name = info.get('name', '?')[:18]
        # Solid cube fill
        ax.bar3d(center[0]-SLOT_SIZE[0]/2, center[1]-SLOT_SIZE[1]/2,
                 center[2]-SLOT_SIZE[2]/2,
                 SLOT_SIZE[0], SLOT_SIZE[1], SLOT_SIZE[2],
                 color=color, alpha=0.85, edgecolor='black', linewidth=0.5)
        # Etiket
        ax.text(center[0], center[1], center[2]+SLOT_SIZE[2]/2+0.05,
                f"S{s_idx+1}\nID#{info.get('id','?')}\n{name}",
                fontsize=6, ha='center', va='bottom',
                color='white',
                bbox=dict(boxstyle='round,pad=0.2', facecolor='black',
                          alpha=0.6, edgecolor='none'))

    # 5) ESP32 sensörler (4M kırmızı + 4S sarı) + UWB master (mavi halka)
    uwb_positions = {tuple(n['position_m']) for n in UWB_NODES}
    for n in ESP32_NODES:
        pos = n['position_m']
        is_master = n['type'] == 'master'
        color = '#e74c3c' if is_master else '#f39c12'
        marker = 's' if is_master else '^'
        ax.scatter([pos[0]], [pos[1]], [pos[2]], c=color, marker=marker,
                   s=40 if is_master else 25, edgecolors='black', linewidths=0.5)
        ax.text(pos[0], pos[1], pos[2]+0.15, n['id'],
                fontsize=6, ha='center', color=color, fontweight='bold')
        # UWB var ise mavi halka ekle
        if tuple(pos) in uwb_positions:
            # Küçük mavi nokta UWB göstergesi (master'ın yanında)
            ax.scatter([pos[0]+0.05], [pos[1]], [pos[2]+0.05], c='#3498db',
                       marker='o', s=20, edgecolors='black', linewidths=0.3)

    # 6) Axes ayarları
    ax.set_xlim(0, ROOM_W)
    ax.set_ylim(0, ROOM_D)
    ax.set_zlim(0, ROOM_H)
    ax.set_xlabel('X (m)', fontsize=7)
    ax.set_ylabel('Y (m)', fontsize=7)
    ax.set_zlabel('Z (m)', fontsize=7)

sample_summary = {}
for i in range(len(test_ds)):
    s = test_ds[i]
    slot_gt = s['labels']['slot'].numpy()
    mat_gt = s['labels']['material'].numpy()
    n_filled = int(slot_gt.sum())
    filled_slots = [int(idx) for idx in np.where(slot_gt == 1)[0]]
    materials = [int(mat_gt[idx]) for idx in filled_slots]
    unique_mats = set(materials)
    sample_summary[i] = {

        'filled_slots': filled_slots,
        'materials': materials,
        'unique_mat_count': len(unique_mats),
    }

# 8 senaryo için en uygun sample'ları seç
def find_sample(predicate, fallback_min_filled=None):
    """Predicate'e uyan ilk sample'ı bul, yoksa fallback."""
    for idx, info in sample_summary.items():
        if predicate(info):
            return idx
    if fallback_min_filled is not None:
        # Fallback: belirli minimum dolu slot'lu ilk sample
        for idx, info in sample_summary.items():
            if info['n_filled'] >= fallback_min_filled:
                return idx
    return list(sample_summary.keys())[0]

     find_sample(lambda i: i['n_filled'] == 1 and i['filled_slots'][0] in [0, 2, 4],
                 fallback_min_filled=1)),
    ('Senaryo 3: Tek dolu (sağ kolon)',
     find_sample(lambda i: i['n_filled'] == 1 and i['filled_slots'][0] in [1, 3, 5],
                 fallback_min_filled=1)),
    ('Senaryo 4: Yarı dolu (sol kolon S1+S3+S5)',

import torch

def model_forward_single(sample):
    """Tek bir sample için model forward + DB lookup → products_by_slot dict."""
    csi = torch.from_numpy(np.asarray(sample['csi'])).float().unsqueeze(0).to(device)
    uwb = torch.from_numpy(np.asarray(sample['uwb'])).float().unsqueeze(0).to(device)
    lg = torch.from_numpy(np.asarray(sample['link_geo'])).float().unsqueeze(0).to(device)
    with torch.no_grad():
        out = model(csi, uwb, lg, return_aux=False)
    det = (torch.sigmoid(out['detection_logits'][0]) > 0.5).cpu().numpy()
    cw_logits = out['barcode_codeword'][0].cpu().numpy()  # (6, 7)
    products = {}
    for s in range(6):
        if not det[s]:
            continue
        bits = (cw_logits[s] > 0).astype(np.uint8)
        data_4bit, corrected = hamming_decode_7_4(bits)
        pid = int(data_bits_to_int(data_4bit))
        info = db.lookup(pid)
        info['hamming_corrected'] = bool(corrected)
        products[s] = info
    return products, det
